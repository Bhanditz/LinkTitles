45  function execute($par) {
46  // Prevent non-authorized users from executing the batch processing.
47  if ( !$this->userCanExecute( $this->getUser() ) ) {
48  $this->displayRestrictionError();
49  return;
50  }
51 
52  $request = $this->getRequest();
53  $output = $this->getOutput();
54  $this->setHeaders();
55 
56  // Determine whether this page was requested via GET or POST.
57  // If GET, display information and a button to start linking.
58  // If POST, start or continue the linking process.
59  if ( $request->wasPosted() ) {
60  if ( array_key_exists('s', $request->getValues()) ) {
61  $this->process($request, $output);
62  }
63  else
64  {
65  $this->buildInfoPage($request, $output);
66  }
67  }
68  else
69  {
70  $this->buildInfoPage($request, $output);
71  }
72  }

81  private function process(&$request, &$output) {
82  global $wgLinkTitlesTimeLimit;
83 
84  // Start the stopwatch
85  $startTime = microtime(true);
86 
87  // Connect to the database
88  $dbr = wfGetDB( DB_SLAVE );
89 
90  // Fetch the start index and max number of records from the POST
91  // request.
92  $postValues = $request->getValues();
93 
94  // Convert the start index to an integer; this helps preventing
95  // SQL injection attacks via forged POST requests.
96  $start = intval($postValues['s']);
97 
98  // If an end index was given, we don't need to query the database
99  if ( array_key_exists('e', $postValues) ) {
100  $end = intval($postValues['e']);
101  }
102  else
103  {
104  // No end index was given. Therefore, count pages now.
105  $end = $this->countPages($dbr);
106  };
107 
108  array_key_exists('r', $postValues) ?
109  $reloads = $postValues['r'] :
110  $reloads = 0;
111 
112  // Retrieve page names from the database.
113  $res = $dbr->select(
114  'page',
115  'page_title',
116  array(
117  'page_namespace = 0',
118  ),
119  __METHOD__,
120  array(
121  'LIMIT' => 999999999,
122  'OFFSET' => $start
123  )
124  );
125 
126  // Iterate through the pages; break if a time limit is exceeded.
127  foreach ( $res as $row ) {
128  $curTitle = $row->page_title;
129  LinkTitles::processPage($curTitle, $this->getContext());
130  $start += 1;
131 
132  // Check if the time limit is exceeded
133  if ( microtime(true)-$startTime > $wgLinkTitlesTimeLimit )
134  {
135  break;
136  }
137  }
138 
139  $this->addProgressInfo($output, $curTitle, $start, $end);
140 
141  // If we have not reached the last page yet, produce code to reload
142  // the extension's special page.
143  if ( $start < $end )
144  {
145  $reloads += 1;
146  // Build a form with hidden values and output JavaScript code that
147  // immediately submits the form in order to continue the process.
148  $output->addHTML($this->getReloaderForm($request->getRequestURL(),
149  $start, $end, $reloads));
150  }
151  else // Last page has been processed
152  {
153  $this->addCompletedInfo($output, $start, $end, $reloads);
154  }
155  }

159  private function buildInfoPage(&$request, &$output) {
160  $url = $request->getRequestURL();
161 
162  // TODO: Put the page contents in messages in the i18n file.
163  $output->addWikiText(
164 <<<EOF
165 LinkTitles extension: http://www.mediawiki.org/wiki/Extension:LinkTitles
166 
167 Source code: http://github.com/bovender/LinkTitles
168 
169 == Batch Linking ==
170 You can start a batch linking process by clicking on the button below.
171 This will go through every page in the normal namespace of your Wiki and
172 insert links automatically. This page will repeatedly reload itself, in
173 order to prevent blocking the server. To interrupt the process, simply
174 close this page.
175 EOF
176  );
177  $output->addHTML(
178 <<<EOF
179 <form method="post" action="${url}">
180  <input type="submit" value="Start linking" />
181  <input type="hidden" name="s" value="0" />
182 </form>
183 EOF
184  );
185  }

193  private function addProgressInfo(&$output, $curTitle, $index, $end) {
194  $progress = $index / $end * 100;
195  $percent = sprintf("%01.1f", $progress);
196 
197  $output->addWikiText(
198 <<<EOF
199 == Processing pages... ==
200 The [http://www.mediawiki.org/wiki/Extension:LinkTitles LinkTitles]
201 extension is currently going through every page of your wiki, adding links to
202 existing pages as appropriate.
203 
204 === Current page: $curTitle ===
205 EOF
206  );
207  $output->addHTML(
208 <<<EOF
209 <p>Page ${index} of ${end}.</p>
210 <div style="width:100%; padding:2px; border:1px solid #000; position: relative;
211  margin-bottom:16px;">
212  <span style="position: absolute; left: 50%; font-weight:bold; color:#555;">
213  ${percent}%
214  </span>
215  <div style="width:${progress}%; background-color:#bbb; height:20px; margin:0;"></div>
216 </div>
217 EOF
218  );
219  $output->addWikiText(
220 <<<EOF
221 === To abort, close this page, or hit the 'Stop' button in your browser ===
222 [[Special:LinkTitles|Return to Special:LinkTitles.]]
223 EOF
224  );
225  }

235  private function getReloaderForm($url, $start, $end, $reloads) {

237 <<<EOF

239  <input type="hidden" name="s" value="${start}" />
240  <input type="hidden" name="e" value="${end}" />
241  <input type="hidden" name="r" value="${reloads}" />

244  document.linktitles.submit();

269 |-

271 |}
