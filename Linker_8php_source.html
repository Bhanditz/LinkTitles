70  public function linkContent( Source $source ) {
71  if ( !$source->canBeLinked() ) {
72  return;
73  }
74 
75  ( $this->config->firstOnly ) ? $limit = 1 : $limit = -1;
76  $limitReached = false;
77  $newLinks = false; // whether or not new links were added
78  $newText = $source->getText();
79  $splitter = Splitter::singleton( $this->config );
80  $targets = Targets::singleton( $source->getTitle(), $this->config );
81 
82  // Iterate through the target page titles
83  foreach( $targets->queryResult as $row ) {
84  $target = new Target( $row->page_namespace, $row->page_title, $this->config );
85 
86  // Don't link current page and don't link if the target page redirects
87  // to the current page or has the __NOAUTOLINKTARGET__ magic word
88  // (as required by the actual LinkTitles configuration).
89  if ( $target->isSameTitle( $source ) || !$target->mayLinkTo( $source ) ) {
90  continue;
91  }
92 
93  // Dealing with existing links if the firstOnly option is set:
94  // A link to the current page should only be recognized if it appears in
95  // clear text, i.e. we do not count piped links as existing links.
96  // (Similarly, by design, redirections should not be counted as existing links.)
97  if ( $limit == 1 && preg_match( '/\[\[' . $target->getCaseSensitiveLinkValueRegex() . ']]/' , $source->getText() ) ) {
98  continue;
99  }
100 
101  // Split the page content by non-linkable sections.
102  // Credits to inhan @ StackOverflow for suggesting preg_split.
103  // See http://stackoverflow.com/questions/10672286
104  $arr = $splitter->split( $newText );
105  $count = 0;
106 
107  // Cache the target title text for the regex callbacks
108  $this->linkValue = $target->getPrefixedTitleText();
109 
110  // Even indexes will point to sections of the text that may be linked
111  for ( $i = 0; $i < count( $arr ); $i += 2 ) {
112  $arr[$i] = preg_replace_callback( $target->getCaseSensitiveRegex(),
113  array( $this, 'simpleModeCallback'),
114  $arr[$i], $limit, $count );
115  if ( $this->config->firstOnly && ( $count > 0 ) ) {
116  $limitReached = true;
117  break;
118  };
119  };
120  if ( $count > 0 ) {
121  $newLinks = true;
122  $newText = implode( '', $arr );
123  }
124 
125  // If smart mode is turned on, the extension will perform a second
126  // pass on the page and add links with aliases where the case does
127  // not match.
128  if ( $this->config->smartMode && !$limitReached ) {
129  if ( $count > 0 ) {
130  // Split the text again because it was changed in the first pass.
131  $arr = $splitter->split( $newText );
132  }
133 
134  for ( $i = 0; $i < count( $arr ); $i+=2 ) {
135  // even indexes will point to text that is not enclosed by brackets
136  $arr[$i] = preg_replace_callback( $target->getCaseInsensitiveRegex(),
137  array( $this, 'smartModeCallback'),
138  $arr[$i], $limit, $count );
139  if ( $this->config->firstOnly && ( $count > 0 )) {
140  break;
141  };
142  };
143  if ( $count > 0 ) {
144  $newLinks = true;
145  $newText = implode( '', $arr );
146  }
147  } // $wgLinkTitlesSmartMode
148  }; // foreach $res as $row
149 
150  if ( $newLinks ) {
151  return $newText;
152  }
153  }

161  private function simpleModeCallback( array $matches ) {
162  // If the link value is longer than the match, it must be prefixed with
163  // a namespace. In this case, we build a piped link.
164  if ( strlen( $this->linkValue ) > strlen( $matches[0] ) ) {
165  return '[[' . $this->linkValue . '|' . $matches[0] . ']]';
166  } else {
167  return '[[' . $matches[0] . ']]';
168  }
169  }

183  private function smartModeCallback( array $matches ) {
184  // If cases of the target page title and the actual occurrence in the text
185  // are not identical, we need to build a piped link.
186  // How case-identity is determined depends on the $wgCapitalLinks setting:
187  // with $wgCapitalLinks = true, the case of first letter of the title is
188  // not significant.
189  if ( $this->config->capitalLinks ) {
190  $needPipe = strcmp( substr( $this->linkValue, 1 ), substr( $matches[ 0 ], 1 ) ) != 0;
191  } else {
192  $needPipe = strcmp( $this->linkValue, $matches[ 0 ] ) != 0;
193  }
194  if ( $needPipe ) {
195  return '[[' . $this->linkValue . '|' . $matches[ 0 ] . ']]';
196  } else {
197  return '[[' . $matches[ 0 ] . ']]';
198  }
199  }
