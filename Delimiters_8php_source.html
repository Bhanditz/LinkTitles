83  * text that should be parsed from text that should not be
84  * parsed (e.g. inside existing links etc.)
85  */
86  private function buildDelimiters() {
87  // Use unicode character properties rather than \b escape sequences
88  // to detect whole words containing non-ASCII characters as well.
89  // Note that this requires a PCRE library that was compiled with
90  // --enable-unicode-properties
91  ( $this->config->wordStartOnly ) ? $this->wordStart = '(?<!\pL)' : $this->wordStart = '';
92  ( $this->config->wordEndOnly ) ? $this->wordEnd = '(?!\pL)' : $this->wordEnd = '';
93 
94  if ( $this->config->skipTemplates )
95  {
96  // Use recursive regex to balance curly braces;
97  // see http://www.regular-expressions.info/recurse.html
98  $templatesDelimiter = '{{(?>[^{}]|(?R))*}}|';
99  } else {
100  // Match template names (ignoring any piped [[]] links in them)
101  // along with the trailing pipe and parameter name or closing
102  // braces; also match sequences of '|wordcharacters=' (without
103  // spaces in them) that usually only occur as parameter names in
104  // transclusions (but could also occur as wiki table cell contents).
105  // TODO: Find a way to match parameter names in transclusions, but
106  // not in table cells or other sequences involving a pipe character
107  // and equal sign.
108  $templatesDelimiter = '{{[^|]*?(?:(?:\[\[[^]]+]])?)[^|]*?(?:\|(?:\w+=)?|(?:}}))|\|\w+=|';
109  }
110 
111  // Build a regular expression that will capture existing wiki links ("[[...]]"),
112  // wiki headings ("= ... =", "== ... ==" etc.),
113  // urls ("http://example.com", "[http://example.com]", "[http://example.com Description]",
114  // and email addresses ("mail@example.com").
115  // Since there is a user option to skip headings, we make this part of the expression
116  // optional. Note that in order to use preg_split(), it is important to have only one
117  // capturing subpattern (which precludes the use of conditional subpatterns).
118  ( $this->config->parseHeadings ) ? $delimiter = '' : $delimiter = '=+.+?=+|';
119  $urlPattern = '[a-z]+?\:\/\/(?:\S+\.)+\S+(?:\/.*)?';
120  $this->splitter = '/(' . // exclude from linking:
121  '\[\[.*?\]\]|' . // links
122  $delimiter . // titles (if requested)
123  $templatesDelimiter . // templates (if requested)
124  '^ .+?\n|\n .+?\n|\n .+?$|^ .+?$|' . // preformatted text
125  '<nowiki>.*?<.nowiki>|<code>.*?<\/code>|' . // nowiki/code
126  '<pre>.*?<\/pre>|<html>.*?<\/html>|' . // pre/html
127  '<script>.*?<\/script>|' . // script
128  '<gallery>.*?<\/gallery>|' . // gallery
129  '<div.+?>|<\/div>|' . // attributes of div elements
130  '<span.+?>|<\/span>|' . // attributes of span elements
131  '<file>[^<]*<\/file>|' . // stuff inside file elements
132  'style=".+?"|class=".+?"|' . // styles and classes (e.g. of wikitables)
133  '<noautolinks>.*?<\/noautolinks>|' . // custom tag 'noautolinks'
134  '\[' . $urlPattern . '\s.+?\]|'. $urlPattern . '(?=\s|$)|' . // urls
135  '(?<=\b)\S+\@(?:\S+\.)+\S+(?=\b)' . // email addresses
136  ')/ismS';
137  }
