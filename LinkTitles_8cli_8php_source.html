73  public function execute() {
74  $index = intval($this->getOption('start', 0));
75  if ( $index < 0 ) {
76  $this->error('FATAL: Start index must be 0 or greater.', 1);
77  };
78 
79  // Connect to the database
80  $dbr = $this->getDB( DB_SLAVE );
81 
82  // Retrieve page names from the database.
83  $res = $dbr->select(
84  'page',
85  'page_title',
86  array(
87  'page_namespace = 0',
88  ),
89  __METHOD__,
90  array(
91  'LIMIT' => 999999999,
92  'OFFSET' => $index
93  )
94  );
95  $numPages = $res->numRows();
96  $context = RequestContext::getMain();
97  $this->output("Processing ${numPages} pages, starting at index ${index}...\n");
98 
99  // Iterate through the pages; break if a time limit is exceeded.
100  foreach ( $res as $row ) {
101  $index += 1;
102  $curTitle = $row->page_title;
103  $this->output(
104  sprintf("\rPage #%d (%02.0f%%)", $index, $index / $numPages * 100)
105  );
106  LinkTitles::processPage($curTitle, $context);
107  }
108 
109  $this->output("Finished parsing.");
110  }
111 }
